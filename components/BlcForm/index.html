Validation schema
const ValidationSchema = Yup.object().shape({
    idTextField: Yup.string()
        .min(2, 'Text too short')
        .max(50, 'Text too long')
        .required('Text required'),
    idTextarea: Yup.string()
        .min(2, 'Textarea too short')
        .max(50, 'Textarea too long')
        .required('Textarea required'),
    idNumeric: Yup.number()
        .min(2, 'Numeric too short')
        .max(10, 'Numeric too long')
        .required('Numeric required'),
    idCheckbox: Yup.boolean()
        .required('Checkbox required'),
    idSelect: Yup.mixed()
        .oneOf(['select value 1', 'select value 2', 'select value 3'])
        .required('select option required'),
    idRadio: Yup.mixed()
        .oneOf(['radio value 1', 'radio value 2', 'radio value 3'])
        .required('ratio option required'),
});
